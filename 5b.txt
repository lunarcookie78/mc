// PROGRAM-5SMALLEST NUMBERINANARRAYOF32-BITNUMBERS
 AREA SMALLEST, CODE, READONLY
 ENTRY
 ; Mark first instruction to execute
 START
 MOVR5,#6
 LDR R1, =VALUE1
 LDR R2, [R1], #4
 LOOP LDRR4,[R1], #4
 CMPR2,R4
 BLS LOOP1
 MOVR2,R4
 LOOP1 SUBS R5, R5, #1
 CMPR5,#0
 BNE LOOP
 LDR R4, =RESULT
 STR R2, [R4]
 ; INTIALIZE COUNTER TO 6(i.e. N=7)
 ; LOADS THE ADDRESS OFFIRST VALUE
 ; WORDALIGNT0ARRAYELEMENT
 ; WORDALIGNT0ARRAYELEMENT
 ; COMPARENUMBERS
 ; IF THE FIRST NUMBER IS < THEN GOTOLOOP1
 ; IF THE FIRST NUMBER IS > THEN MOV
 ; CONTENT R4TOR2
 ; DECREMENT COUNTER
 ; COMPARECOUNTERTO0
 ; LOOP BACKTILL ARRAY ENDS
 ; LOADS THE ADDRESS OFRESULT
 ; STORES THE RESULT FROM R2INLOCATION
 ;RESULT
 XSS B XSS
 ; ARRAY OF 32BIT NUMBERS(N=7)
 VALUE1 DCD0X44444444, 0X22222222, 0X11111111, 0X33333333
 DCD 0XAAAAAAA,0X88888888, 0X99999999
 AREA DATA2, DATA, READWRITE
 ; TO STORE RESULT IN GIVEN ADDRESS
 RESULT DCD 0X0
 END
 ; Mark end of file

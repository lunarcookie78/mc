 ;PROGRAM-6 COUNTINGNUMBEROFZEROESANDONESIN32bitDATA
 AREA ONEZERO, CODE, READONLY
 ENTRY
 ; Mark first instruction to execute
 START
 LOOP
 LOOP0
 MOVR2,#0
 MOVR3,#0
 MOVR7,#2
 LDR R6, =VALUE
 MOVR1,#32
 LDR R0, [R6], #4
 ; COUNTER FORONES
 ; COUNTER FORZEROS
 ; COUNTER TOGETTWO32-bitnumbers
 ; LOADS THE ADDRESS OFVALUE
 ; 32 BITS COUNTER
 ; GET THE 32BIT VALUE
 MOVSR0,R0, ROR#1 ;RIGHTSHIFTTOCHECKCARRYBIT(1's/0's)
BHI ONES
 ; IF CARRY BIT IS 1 GOTO ONES BRANCH
 ; OTHERWISE NEXT
 ZEROS ADDR3,R3,#1
 B LOOP1
 ONES ADDR2,R2,#1
 LOOP1
 SUBS R1, R1, #1
 BNE LOOP0
 SUBS R7, R7, #1
 CMPR7,#0
 BNE LOOP
 XSS BXSS
 VALUE DCD0X3, 0X2
 END
 ; IF CARRY BIT IS 0 THEN INCREMENT THE
 ; COUNTER BY1(R3)
 ; BRANCHTOLOOP1
 ; IF CARRY BIT IS 1 THEN INCREMENT THE
 ; COUNTER BY1(R2)
 ; COUNTER VALUE DECREMENTEDBY1
 ; IF NOT EQUAL GOTOTOLOOP0CHECKS32BIT
 ; COUNTER VALUE DECREMENTEDBY1
 ; COMPARECOUNTERR7TO0
 ; IF NOT EQUAL GOTOTOLOOP
 ; TWOVALUESINANARRAY
 ; Mark end of file
